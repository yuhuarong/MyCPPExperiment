## Date类

```C++
class Date{
	public:
		void init();
		void init(int, int, int);
		int diff(const Date& date);
		
		void setYear(int);
		void setMonth(int);
		int setDay(int);
		
		int getYear() const;
		int getMonth() const;
		int getDay() const;
		int getWeek();
		
		friend ostream& operator<<(ostream& out, const Date& date);
		friend istream& operator>>(istream& in, Date& date);
		
		int operator==(const Date& date);
		int operator>(const Date& date);
		int operator<(const Date& date);

		int year;
		int month;
		int day;
		
		int isLeap();
		int getMax();
		
		Date next(const Date&);
		
}; 

```

## Contract类

```C++
class Contract{
	public:
		Contract();
		
		Contract(int id, string partyA, string partyB, string content, 
                    Date validStart, Date ValidEnd, double money, Date signedDate);
		
		void clear();
		
		friend ostream& operator<<(ostream& out, const Contract& contract);
		friend istream& operator>>(istream& in, Contract& contract);

		int id;
		string partyA;
		string partyB;
		string content;
		Date validStart;
		Date validEnd;
		double money;
		Date signedDate;
};
```

## ContractUI
