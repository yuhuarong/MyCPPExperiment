## Contract类

```C++
class Contract{
	public:
		Contract();
		
		Contract(int id, string partyA, string partyB, string content, 
                    Date validStart, Date ValidEnd, double money, Date signedDate);
		
		void clear();
		
		friend ostream& operator<<(ostream& out, const Contract& contract);
		friend istream& operator>>(istream& in, Contract& contract);

		int id;
		string partyA;
		string partyB;
		string content;
		Date validStart;
		Date validEnd;
		double money;
		Date signedDate;
};
```

## ContractUI
```C++
class ContractUI: public Contract{
	public: 
		ContractUI():Contract(){}
		ContractUI(int id, string partyA, string partyB, string content, 
                        Date validStart, Date ValidEnd, double money, Date signedDate):Contract(id, partyA, partyB, 
                        content, validStart, validEnd, money, signedDate){}
		
		ContractUI& operator=(const Contract& c);
		friend ostream& operator<<(ostream& out, const ContractUI& contract);
		friend istream& operator>>(istream& in, ContractUI& contract);
}; 
```

```C++
ContractUI& operator=(const Contract& c){
	this->id=c.id;
	this->partyA=c.partyA;
	this->partyB=c.partyB;
	this->content=c.content;
	this->validStart=c.validStart;
	this->validEnd=c.validEnd;
	this->money=c.money;
	this->signedDate=c.signedDate;
	return *this;
}
```
